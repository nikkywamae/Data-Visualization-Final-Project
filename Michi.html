from numpy import size
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches

df = pd.read_csv("Voter_demographics_tables.csv")
df = df[df["County"].str.strip().str.lower() != "total"].copy()
df["County"] = df["County"].str.strip()

# Utilized AI to organize data from census bureau
washington_county_population_2020 = {
    "Adams": 20613, "Asotin": 22285, "Benton": 206873,
    "Chelan": 79074, "Clallam": 77155, "Clark": 503311,
    "Columbia": 3952, "Cowlitz": 110730, "Douglas": 42938,
    "Ferry": 7178, "Franklin": 96749, "Garfield": 2286,
    "Grant": 99123, "Grays Harbor": 75636, "Island": 86857,
    "Jefferson": 32977, "King": 2269675, "Kitsap": 275611,
    "Kittitas": 44337, "Klickitat": 22735, "Lewis": 82149,
    "Lincoln": 10876, "Mason": 65726, "Okanogan": 42104,
    "Pacific": 23365, "Pend Oreille": 13401, "Pierce": 921130,
    "San Juan": 17788, "Skagit": 129523, "Skamania": 12036,
    "Snohomish": 827957, "Spokane": 539339, "Stevens": 46445,
    "Thurston": 294793, "Wahkiakum": 4422, "Walla Walla": 62584,
    "Whatcom": 226847, "Whitman": 47973, "Yakima": 256728,
}
pop_df = pd.DataFrame(list(washington_county_population_2020.items()), columns=["County", "Population"])
df = df.merge(pop_df, on="County", how="left")
df["ActiveRate"] = df["Active"] / df["Population"]
df = df.sort_values("ActiveRate", ascending=True).reset_index(drop=True)
TITLE_FONT = {"fontsize": 14, "fontweight": "bold", "fontfamily": "serif"}
STATE_AVG = df["ActiveRate"].mean()
colors = ["#1560a8" if v >= STATE_AVG else "#a3c9ee" for v in df["ActiveRate"]]
dem_counties = {"King", "Pierce", "Snohomish", "Clark", "Thurston", "Kitsap", "Whatcom", "Skagit", "Island", "Clallam", "Whitman", "Jefferson", "San Juan"}
fig, ax = plt.subplots(figsize=(12, 16))
fig.patch.set_facecolor("white")
bars = ax.barh(df["County"], df["ActiveRate"], color=colors, edgecolor="white", height=0.6)
ax.axvline(STATE_AVG, color="#999999", linewidth=0.8, linestyle="--", zorder=5)
ax.text(STATE_AVG + 0.002, 0.5, f"avg {STATE_AVG:.1%}", color="#999999", fontsize=8, fontfamily="serif", va="bottom")
for bar, val in zip(bars, df["ActiveRate"]):
    ax.text(val + 0.002, bar.get_y() + bar.get_height() / 2,
            f"{val:.1%}", va="center", fontsize=12,
            color="#222222", fontfamily="serif")
for label in ax.get_yticklabels():
    county_name = label.get_text()
    if county_name in dem_counties:
        label.set_color("blue")
        label.set_fontweight("bold")
        label.set_fontsize(20)
    else:
        label.set_color("red")
        label.set_fontweight("bold")
        label.set_fontsize(20)
above = mpatches.Patch(color="#1560a8", label="Above state average (Participation)")
below = mpatches.Patch(color="#a3c9ee", label="Below state average (Participation)")
ax.legend(handles=[above, below], fontsize=9, loc="lower right", framealpha=0.9, edgecolor="#cccccc")
ax.set_xlim(0, df["ActiveRate"].max() + 0.15)
ax.xaxis.set_major_formatter(plt.FuncFormatter(lambda x, _: f"{x:.0%}"))
ax.set_xlabel("Active Registered Voters / Total Population (2020 Census)", fontsize=20,)
ax.tick_params(axis="both", labelsize=9)
ax.spines[["top", "right", "left"]].set_visible(False)
ax.tick_params(axis="y", length=0)
ax.grid(axis="x", linestyle="--", linewidth=0.4, alpha=0.4)
ax.set_title("Active Voter Participation vs. Total Population by County\nWashington State 2024 vs. 2020 Census", size = 35)
plt.tight_layout()
plt.savefig("wa_active_vs_population_colored.png", dpi=300, bbox_inches="tight")

import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
import numpy as np


df_pres = pd.read_csv("cleaned_wa_2024_presidential.csv")
df_pres["County"] = df_pres["County"].astype(str).str.replace(" County", "", regex=False).str.strip()
dem_df = df_pres[df_pres["Party"].astype(str).str.contains("Dem", case=False, na=False)].copy()
rep_df = df_pres[df_pres["Party"].astype(str).str.contains("Rep", case=False, na=False)].copy()
dem_df = dem_df[["County", "Votes"]].rename(columns={"Votes": "Dem_Votes"})
rep_df = rep_df[["County", "Votes"]].rename(columns={"Votes": "Rep_Votes"})
plot_df = dem_df.merge(rep_df, on="County", how="inner")
plot_df["Total_Votes"] = plot_df["Dem_Votes"] + plot_df["Rep_Votes"]
plot_df["Margin"] = abs(plot_df["Dem_Votes"] - plot_df["Rep_Votes"]) / plot_df["Total_Votes"]
plot_df["Winner"] = np.where(plot_df["Dem_Votes"] > plot_df["Rep_Votes"], "Democrat", "Republican")
plot_df["Color"] = plot_df["Winner"].map({"Democrat": "#1560a8", "Republican": "#d62728"})
med_dem_margin = plot_df[plot_df["Winner"] == "Democrat"]["Margin"].mean()
med_rep_margin = plot_df[plot_df["Winner"] == "Republican"]["Margin"].mean()
plot_df["Sort_Margin"] = plot_df.apply(lambda row: row["Margin"] if row["Winner"] == "Democrat" else -row["Margin"], axis=1)
plot_df = plot_df.sort_values("Sort_Margin", ascending=False).reset_index(drop=True)
fig, ax = plt.subplots(figsize=(16, 8))
fig.patch.set_facecolor("white")
bars = ax.bar(plot_df["County"], plot_df["Margin"], color=plot_df["Color"], edgecolor="white", zorder=3)
ax.axhline(med_dem_margin, color="#0b365e", linestyle="--", linewidth=2.5, zorder=4)
ax.axhline(med_rep_margin, color="#801515", linestyle="--", linewidth=2.5, zorder=4)
x_pos = (len(plot_df) / 2) - 4

ax.text(x_pos, med_dem_margin - 0.005, f"Average Dem Win Margin: {med_dem_margin:.1%}", color="black", fontweight="bold", fontfamily="serif", ha="center", va="top", fontsize=15)
ax.text(x_pos, med_rep_margin + 0.005, f"Average GOP Win Margin: {med_rep_margin:.1%}", color="black", fontweight="bold", fontfamily="serif", ha="center", va="bottom", fontsize=15)
ax.set_title("Republicans Have Higher Average Win Margins than Democrats\n2024 Presidential Election Win Margin by Washington County",
             fontsize=25, fontweight="bold", fontfamily="serif", pad=15)
ax.set_ylabel("Win Margin (%)", fontsize=12, fontfamily="serif", fontweight="bold")
ax.set_xlabel("County (Sorted Bluest to Reddest)", fontsize=12, fontfamily="serif", fontweight="bold")
ax.yaxis.set_major_formatter(plt.FuncFormatter(lambda y, _: f"{y:.0%}"))
plt.xticks(rotation=90, fontsize=10)
dem_patch = mpatches.Patch(color="#1560a8", label="Democrat Won")
rep_patch = mpatches.Patch(color="#d62728", label="Republican Won")
ax.legend(handles=[dem_patch, rep_patch], loc="upper right", fontsize=11, framealpha=0.9)
ax.grid(axis="y", linestyle="--", linewidth=0.5, alpha=0.7, zorder=0)
ax.spines[["top", "right"]].set_visible(False)
plt.tight_layout()
plt.show()
