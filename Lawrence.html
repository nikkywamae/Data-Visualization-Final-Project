import pandas as pd
import geopandas as gpd
import matplotlib.pyplot as plt
import matplotlib.colors as mcolors
import matplotlib.cm as cm
import requests, zipfile, os

DEM_COLOR = "#0015BC"
REP_COLOR = "#E9141D"

raw = pd.read_csv("20241105_allcounties.csv")
pres = raw[raw["Race"] == "United States President/Vice President"].copy()
dem = pres[pres["Party"] == "Democratic Party Nominees"][["County", "Votes"]].rename(columns={"Votes": "Dem"})
rep = pres[pres["Party"] == "Republican Party Nominees"][["County", "Votes"]].rename(columns={"Votes": "Rep"})
df = dem.merge(rep, on="County")
df["Total2P"] = df["Dem"] + df["Rep"]
df["DemShare"] = df["Dem"] / df["Total2P"]
df["County"] = df["County"].str.strip()

if not os.path.exists("counties_data/tl_2023_us_county.shp"):
    r = requests.get("https://www2.census.gov/geo/tiger/TIGER2023/COUNTY/tl_2023_us_county.zip")
    with open("us_counties.zip", "wb") as f: f.write(r.content)
    with zipfile.ZipFile("us_counties.zip", "r") as z: z.extractall("counties_data")

wa_base = gpd.read_file("counties_data/tl_2023_us_county.shp")
wa_base = wa_base[wa_base["STATEFP"] == "53"].copy()
wa_base["NAME"] = wa_base["NAME"].str.strip()
wa = wa_base.merge(df, left_on="NAME", right_on="County", how="left")

LABEL_FONT = {"fontsize": 6, "fontweight": "bold"}
cmap = mcolors.LinearSegmentedColormap.from_list("ElectionMap", [REP_COLOR, "#6a3d9a", DEM_COLOR])
norm = mcolors.TwoSlopeNorm(vmin=0.0, vcenter=0.5, vmax=1.0)

fig, ax = plt.subplots(figsize=(13, 9))
fig.patch.set_facecolor("white")
wa.plot(column="DemShare", ax=ax, cmap=cmap, norm=norm, edgecolor="white", linewidth=0.8, missing_kwds={"color": "#cccccc"})

for _, row in wa.iterrows():
    if pd.isna(row["DemShare"]): text_color = "#444444"
    else:
        rgba = cmap(norm(row["DemShare"]))
        brightness = 0.299*rgba[0] + 0.587*rgba[1] + 0.114*rgba[2]
        text_color = "white" if brightness < 0.65 else "#222222"
    ax.annotate(row["NAME"], xy=(row.geometry.centroid.x, row.geometry.centroid.y),
                ha="center", va="center", color=text_color, **LABEL_FONT)

sm = cm.ScalarMappable(cmap=cmap, norm=norm)
sm.set_array([])
cbar = fig.colorbar(sm, ax=ax, fraction=0.03, pad=0.02, ticks=[0, 0.25, 0.5, 0.75, 1.0])
cbar.ax.set_yticklabels(["100% R", "75% R", "50 / 50", "75% D", "100% D"], fontsize=7)
cbar.set_label("Two-Party Vote Share", fontsize=8, fontfamily="sans-serif")

ax.set_title("Eastern Washington Votes Overwhelmingly Red",
             fontsize=13, fontweight="bold", fontfamily="sans-serif", pad=5)
fig.text(0.5, 0.85, "2024 Presidential Election — Vote Share by County, Washington State",
         ha="center", fontsize=10, color="#555555", fontfamily="sans-serif")

ax.set_axis_off()
plt.subplots_adjust(top=0.95)
plt.savefig("wa_election_voteshare.png", dpi=180, bbox_inches="tight")
plt.show()

import pandas as pd
import altair as alt

DEM_COLOR = "#0015BC"
REP_COLOR = "#E9141D"

df = pd.read_csv("cleaned_wa_2024_presidential.csv")

harris = df[df["Candidate"] == "Harris"][["County", "Votes", "Pct_of_Total_Votes"]].rename(
    columns={"Votes": "Harris_Votes", "Pct_of_Total_Votes": "Harris_Pct"})
trump = df[df["Candidate"] == "Trump"][["County", "Votes", "Pct_of_Total_Votes"]].rename(
    columns={"Votes": "Trump_Votes", "Pct_of_Total_Votes": "Trump_Pct"})

df_plot = harris.merge(trump, on="County")
df_plot["Winner"] = df_plot.apply(
    lambda r: "Democrat" if r["Harris_Pct"] > r["Trump_Pct"] else "Republican", axis=1
)
df_plot["Trump_Plot"] = df_plot["Trump_Pct"] * -1

# West-to-east county ordering generated with Generative AI assistance
west_to_east = [
    "Clallam", "Jefferson", "Grays Harbor", "Pacific", "Wahkiakum",
    "Whatcom", "Skagit", "Island", "San Juan", "Snohomish", "King",
    "Pierce", "Thurston", "Mason", "Kitsap", "Lewis", "Cowlitz", "Clark",
    "Skamania", "Klickitat", "Yakima", "Kittitas", "Chelan",
    "Okanogan", "Douglas", "Grant", "Benton", "Franklin", "Adams",
    "Walla Walla", "Columbia", "Garfield", "Asotin", "Whitman", "Lincoln",
    "Spokane", "Ferry", "Stevens", "Pend Oreille"
]

df_plot["County"] = pd.Categorical(df_plot["County"], categories=west_to_east, ordered=True)
df_plot = df_plot.sort_values("County").reset_index(drop=True)

dem_df = df_plot[["County", "Harris_Pct", "Winner"]].copy()
dem_df["Party"] = "Democrat"
dem_df["Pct"] = dem_df["Harris_Pct"]
dem_df["PlotPct"] = dem_df["Harris_Pct"]

rep_df = df_plot[["County", "Trump_Pct", "Winner"]].copy()
rep_df["Party"] = "Republican"
rep_df["Pct"] = rep_df["Trump_Pct"]
rep_df["PlotPct"] = rep_df["Trump_Pct"] * -1

long_df = pd.concat([
    dem_df[["County", "Party", "Pct", "PlotPct", "Winner"]],
    rep_df[["County", "Party", "Pct", "PlotPct", "Winner"]]
]).reset_index(drop=True)

dem_counties = df_plot[df_plot["Winner"] == "Democrat"][["County", "Winner"]].copy()
rep_counties = df_plot[df_plot["Winner"] == "Republican"][["County", "Winner"]].copy()

bars = alt.Chart(long_df).mark_bar(size=10).encode(
    alt.Y("County:N",
          sort=west_to_east,
          axis=alt.Axis(labels=False, ticks=False, title="Counties in Washington (West to East)")),
    alt.X("PlotPct:Q",
          title="Vote Share (%)",
          scale=alt.Scale(domain=[-85, 85]),
          axis=alt.Axis(
              values=[-80, -60, -40, -20, 0, 20, 40, 60, 80],
              labelExpr="abs(datum.value) + '%'",
              labelFontSize=9,
              gridColor="#e0e0e0"
          )),
    alt.Color("Party:N",
              scale=alt.Scale(
                  domain=["Democrat", "Republican"],
                  range=[DEM_COLOR, REP_COLOR]
              ),
              legend=alt.Legend(title="Party", labelFontSize=10, titleFontSize=11)),
    alt.Opacity("Pct:Q",
                scale=alt.Scale(domain=[20, 80], range=[0.5, 1.0]),
                legend=alt.Legend(
                    title="Vote Share Strength",
                    labelFontSize=9,
                    titleFontSize=10,
                    values=[20, 40, 60, 80],
                    labelExpr="datum.value + '%'"
                ))
).properties(width=620, height=800)

zeroline = alt.Chart(pd.DataFrame({"x": [0]})).mark_rule(
    color="black", strokeWidth=2
).encode(x="x:Q")

dem_labels = alt.Chart(dem_counties).mark_text(

).encode(
    alt.Y("County:N", sort=west_to_east, axis=None),
    text="County:N"
).properties(width=120, height=800)

rep_labels = alt.Chart(rep_counties).mark_text(
    align="right", fontSize=9, color=REP_COLOR, fontWeight="bold"
).encode(
    alt.Y("County:N", sort=west_to_east, axis=None),
    text="County:N"
).properties(width=120, height=800)

barchart = alt.hconcat(
    (dem_labels + rep_labels),
    (bars + zeroline),
    spacing=0
).properties(
    title=alt.TitleParams(
        text="Eastern Washington Votes Overwhelmingly Republican",
        subtitle="Two-party vote share by county, west to east | Source: WA Secretary of State, Nov 2024",
        fontSize=15,
        subtitleFontSize=10,
        subtitleColor="#555555",
        anchor="middle"
    )
).configure_view(
    stroke=None
).configure_axis(
    labelFontSize=9,
    titleFontSize=11
)

scatter = alt.Chart(df).mark_point(size=80, opacity=0.85, filled=True).encode(

)

).encode(x="x:Q", y="y:Q", text="text:N")

)

).configure_view(

    labelFontSize=10,
