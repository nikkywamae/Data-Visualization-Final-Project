Chloropleth for County Voting (all ages)¶

In [30]:
import pandas as pd
import geopandas as gpd
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
import requests
import zipfile

df = df[df["County"].str.strip().str.lower() != "total"]

r = requests.get("https://www2.census.gov/geo/tiger/TIGER2023/COUNTY/tl_2023_us_county.zip")
with open("us_counties.zip", "wb") as f:
    f.write(r.content)
with zipfile.ZipFile("us_counties.zip", "r") as z:
    z.extractall("counties_data")

wa = gpd.read_file("counties_data/tl_2023_us_county.shp")
wa = wa[wa["STATEFP"] == "53"].merge(df[["County", "Total"]], left_on="NAME", right_on="County", how="left")

bins = [0, 10_000, 50_000, 100_000, 250_000, 600_000, float("inf")]
labels = ["< 10K", "10K–50K", "50K–100K", "100K–250K", "250K–600K", "600K+"]
colors = ["#dce9f5", "#a8c8e8", "#6aaed6", "#3182bd", "#1560a8", "#08306b"]
wa["bin"] = pd.cut(wa["Total"], bins=bins, labels=labels)

fig, ax = plt.subplots(figsize=(10, 7))
for label, color in zip(labels, colors):
    wa[wa["bin"] == label].plot(ax=ax, color=color, edgecolor="white", linewidth=0.8)
for _, row in wa.iterrows():
    ax.annotate(row["NAME"], xy=(row.geometry.centroid.x, row.geometry.centroid.y),
                ha="center", va="center", fontsize=5.5, color="#222")

patches = [mpatches.Patch(color=c, label=l) for c, l in zip(colors, labels)]
ax.legend(handles=patches, title="Registered Voters", title_fontsize=9, fontsize=8, loc="lower left")
ax.set_title("Washington State Voter Registration by County", fontsize=13, fontweight="bold", pad=12)
ax.set_axis_off()
plt.tight_layout()
plt.show()

import pandas as pd
import altair as alt

df = df[df["County"].str.strip().str.lower() != "total"]

age_cols = ["18-24", "25-34", "35-44", "45-54", "55-64", "65 and over"]
top5 = df.nlargest(5, "Total")
df_melted = top5.melt(id_vars="County", value_vars=age_cols, var_name="Age Group", value_name="Voters")

county_order = top5.sort_values("Total", ascending=False)["County"].tolist()
color_scale = alt.Scale(
    domain=age_cols,
    range=["#c6dbef", "#9ecae1", "#6baed6", "#3182bd", "#1560a8", "#08306b"]
)

alt.Chart(df_melted).mark_bar().encode(
    alt.X("Voters:Q", title="Registered Voters",
          axis=alt.Axis(format="~s", labelFontSize=14, titleFontSize=16)),
    alt.Y("Age Group:N", title=None,
          axis=alt.Axis(labelFontSize=14)),
    alt.Color("Age Group:N", scale=color_scale,
              legend=alt.Legend(title="Age Group", titleFontSize=15, labelFontSize=14,
                                orient="bottom", direction="horizontal",
                                symbolType="square", symbolSize=180)),
    tooltip=["County:N", "Age Group:N", alt.Tooltip("Voters:Q", format=",")]
).properties(
    width=500,
    height=220
).facet(
    facet=alt.Facet("County:N", sort=county_order, title=None,
                    header=alt.Header(labelFontSize=18, labelFontWeight="bold")),
    columns=1
).properties(
    title=alt.TitleParams("Top 5 Washington Counties — Voters by Age Group", fontSize=22, fontWeight="bold")
).resolve_scale(x="independent")
